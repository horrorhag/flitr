123  void process(T * const dataWrite, T const * const dataRead,
124  const size_t componentsPerLine, const size_t height)
125  {
126 
127  //Could be line parallel!
128  for (size_t y=0; y<height; y++)
129  {
130  const size_t lineOffset=y * componentsPerLine;
131 
132  double *lineSum = lineSumArray_ + y;
133  *lineSum=0.0;
134 
135  for (size_t compNum=0; compNum<componentsPerLine; compNum++)
136  {
137  (*lineSum)+=dataRead[lineOffset + compNum];
138  }
139  }
140 
141 
142  double imageSum=0.0;
143 
144  //Single threaded!
145  for (size_t y=0; y<height; y++)
146  {
147  imageSum+=lineSumArray_[y];
148  }
149 
150 
151  const size_t componentsPerImage=componentsPerLine*height;
152  const float average = imageSum / ((double)componentsPerImage);
153  const float eScale=targetAverage_ / average;
154 
155 
156  //Could be pixel parallel.
157  for (size_t y=0; y<height; ++y)
158  {
159  const size_t lineOffset=y * componentsPerLine;
160 
161  for (size_t compNum=0; compNum<componentsPerLine; ++compNum)
162  {
163  dataWrite[lineOffset + compNum]=dataRead[lineOffset + compNum] * eScale;
164  }
165  }
166  }

272  void process(T * const dataWrite, T const * const dataRead,
273  const size_t width, const size_t height)
274  {
275  const size_t halfWindowSize=windowSize_>>1;
276  const size_t widthMinusWindowSize=width - windowSize_;
277  const size_t heightMinusWindowSize=height - windowSize_;
278  const double recipWindowSizeSquared=1.0f / float(windowSize_*windowSize_);
279 
280  for (size_t y=1; y<heightMinusWindowSize; ++y)
281  {
282  const size_t lineOffsetUS=y * width;
283  const size_t lineOffsetDS=(y+halfWindowSize) * width + halfWindowSize;
284 
285  for (size_t x=1; x<widthMinusWindowSize; ++x)
286  {
287  const double windowSum=integralImageData_[(lineOffsetUS+x) + (windowSize_-1) + (windowSize_-1)*width]
288  - integralImageData_[(lineOffsetUS+x) + (windowSize_-1) - width]
289  - integralImageData_[(lineOffsetUS+x) - 1 + (windowSize_-1)*width]
290  + integralImageData_[(lineOffsetUS+x) - 1 - width];
291 
292  const float windowAvrg=float(windowSum * recipWindowSizeSquared);
293 
294  const float eScale=targetAverage_ / windowAvrg;
295 
296  dataWrite[lineOffsetDS + x]=dataRead[lineOffsetDS + x] * eScale;
297  }

305  const size_t halfWindowSize=windowSize_>>1;
306  const size_t widthMinusWindowSize=width - windowSize_;
307  const size_t heightMinusWindowSize=height - windowSize_;
308  const double recipWindowSizeSquared=1.0f / float(windowSize_*windowSize_);
309 
310  for (size_t y=1; y<heightMinusWindowSize; ++y)
311  {
312  size_t lineOffsetTL=((y-1)*width + 0) * 3;
313  size_t lineOffsetTR=((y-1)*width + windowSize_) * 3;
314  size_t lineOffsetBL=((y-1)*width + windowSize_*width) * 3;
315  size_t lineOffsetBR=((y-1)*width + windowSize_*width + windowSize_) * 3;
316 
317  size_t lineOffset=((y+halfWindowSize) * width + halfWindowSize) * 3 + 3;
318 
319  for (size_t x=1; x<widthMinusWindowSize; ++x)
320  {
321  const double windowSumR=integralImageData_[lineOffsetBR + 0]
322  - integralImageData_[lineOffsetBL + 0]
323  - integralImageData_[lineOffsetTR + 0]
324  + integralImageData_[lineOffsetTL + 0];
325 
326  const double windowSumG=integralImageData_[lineOffsetBR + 1]
327  - integralImageData_[lineOffsetBL + 1]
328  - integralImageData_[lineOffsetTR + 1]
329  + integralImageData_[lineOffsetTL + 1];
330 
331  const double windowSumB=integralImageData_[lineOffsetBR + 2]
332  - integralImageData_[lineOffsetBL + 2]
333  - integralImageData_[lineOffsetTR + 2]
334  + integralImageData_[lineOffsetTL + 2];
335 
336  const float windowAvrgR=float(windowSumR * recipWindowSizeSquared);
337  const float windowAvrgG=float(windowSumG * recipWindowSizeSquared);
338  const float windowAvrgB=float(windowSumB * recipWindowSizeSquared);
